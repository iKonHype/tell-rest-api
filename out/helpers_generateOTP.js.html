/**
 * @module helpers/generateOTP
 */

const generateOTP = () => {
  var digits = "0123456789";
  let OTP = "";
  for (let i = 0; i < 6; i++) {
    OTP += digits[Math.floor(Math.random() * 10)];
  }
  console.log("OTP code: " + OTP);
  return OTP;
};

module.exports = generateOTP;
